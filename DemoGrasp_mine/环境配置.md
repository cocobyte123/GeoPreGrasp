### 安装uv

```
python3 -m pip install --user uv         # 使用pip安装uv
~/.local/bin/uv --version                # 查看uv是否安装在此环境下
echo 'export PATH="$HOME/.local/bin:$PATH"' >> ~/.bashrc     # 添加包含uv的环境变量，否则无法使用
```

### 创建环境 & 安装依赖

```
uv venv --python 3.8.19 demograsp
source demograsp/bin/activate

# 安装pytorch
uv pip install torch==2.4.0 torchvision==0.19.0 torchaudio==2.4.0 --index-url https://download.pytorch.org/whl/cu124

# 给环境安装pip
uv pip install pip setuptools wheel

# 安装chumpy
uv pip install chumpy==0.70 --no-build-isolation
```

### 安装gym
```
mkdir thirdparty
cd thirdparty

# 安装isaacgym
tar -zxvf IsaacGym_Preview_4_Package.tar.gz 
cd isaacgym/python/
uv pip install -e .
cd ../..

# 安装IsaacGymEnv
git clone git@github.com:isaac-sim/IsaacGymEnvs.git
cd IsaacGymEnvs/
uv pip install -e .
```
